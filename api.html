Quick Start

// Fetch all AI model prices
const res = await fetch('https://getapipulse.com/api/v1/pricing.json');
const data = await res.json();

// Find cheapest model by input price
const cheapest = data.models
  .filter(m => !m.deprecated)
  .sort((a, b) => a.input - b.input)[0];

console.log(`Cheapest: ${cheapest.name} at $${cheapest.input}/1M tokens`);

// Calculate cost for a workload
function estimateCost(model, inputTokens, outputTokens) {
  return (inputTokens / 1e6) * model.input
       + (outputTokens / 1e6) * model.output;
}

// Compare GPT-5 vs Claude Sonnet 4.6
const gpt5 = data.models.find(m => m.id === 'openai-gpt5');
const sonnet = data.models.find(m => m.id === 'anthropic-sonnet46');
console.log('GPT-5:', estimateCost(gpt5, 1e6, 5e6).toFixed(2));
console.log('Sonnet:', estimateCost(sonnet, 1e6, 5e6).toFixed(2));
import requests

# Fetch all AI model prices
data = requests.get(
    "https://getapipulse.com/api/v1/pricing.json"
).json()

# Find cheapest non-deprecated model
active = [m for m in data["models"] if not m["deprecated"]]
cheapest = min(active, key=lambda m: m["input"])
print(f"Cheapest: {cheapest['name']} at ${cheapest['input']}/1M tokens")

# Calculate cost for a workload
def estimate_cost(model, input_tokens, output_tokens):
    return (input_tokens / 1e6) * model["input"] \
         + (output_tokens / 1e6) * model["output"]

# Compare GPT-5 vs Claude Sonnet 4.6
gpt5 = next(m for m in data["models"] if m["id"] == "openai-gpt5")
sonnet = next(m for m in data["models"] if m["id"] == "anthropic-sonnet46")
print(f"GPT-5: ${estimate_cost(gpt5, 1e6, 5e6):.2f}")
print(f"Sonnet: ${estimate_cost(sonnet, 1e6, 5e6):.2f}")
# Fetch all pricing data
curl https://getapipulse.com/api/v1/pricing.json

# Pretty-print with jq
curl -s https://getapipulse.com/api/v1/pricing.json | jq .

# Get only model names and prices
curl -s https://getapipulse.com/api/v1/pricing.json | \
  jq '.models[] | {name, provider, input, output}'

# Filter by provider
curl -s https://getapipulse.com/api/v1/pricing.json | \
  jq '.models[] | select(.provider == "openai")'

# Find cheapest model
curl -s https://getapipulse.com/api/v1/pricing.json | \
  jq '[.models[] | select(.deprecated == false)] | sort_by(.input) | first'

Model Fields

Field Type Description
id string Unique model identifier (e.g., openai-gpt5)
name string Display name (e.g., GPT-5)
provider string Provider ID (e.g., openai). Maps to providers[].id
tier string One of: premium, mid, budget
input number Input price per 1M tokens (USD)
output number Output price per 1M tokens (USD)
context string Context window size (e.g., 272K, 1M)
deprecated boolean true if the model is deprecated or sunsetting
replacement string? ID of the recommended replacement model (only on deprecated models)

Frequently Asked Questions

Is this API really free? +
Yes. No API key, no rate limits, no signup. It's a static JSON file hosted on Vercel's edge network. We update it monthly when providers change pricing.
Can I use this in my commercial product? +
Yes. Licensed under CC-BY-4.0. You can use it in dashboards, calculators, and SaaS products. Attribution with a link to getapipulse.com is required.
How often is the data updated? +
Monthly, or whenever providers announce price changes. The meta.updated field shows the last update date. Each model has a verified date.
What about rate limits? +
None. It's a static JSON file served by Vercel's CDN. You can fetch it as often as you need. We recommend caching the response and refreshing daily.
